<script>
    const supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY);
    
    let currentUser = null;
    let currentVehicle = null;
    let active = true;
    let selectedMessageId = null;

    // تهيئة الصفحة
    async function initDashboard() {
        // 1. التحقق من تسجيل الدخول
        const { data: { user }, error } = await supabase.auth.getUser();
        if (error || !user) {
            alert('الرجاء تسجيل الدخول أولاً');
            window.location.href = '/login.html';
            return;
        }
        currentUser = user;
        
        // 2. جلب بيانات الملف الشخصي
        const { data: profile } = await supabase
            .from('profiles')
            .select('phone')
            .eq('id', user.id)
            .single();
        if (profile) {
            document.getElementById('phoneDisplay').innerText = profile.phone || 'غير مسجل';
        }

        // 3. جلب مركبات المستخدم
        const { data: vehicles } = await supabase
            .from('vehicles')
            .select('*')
            .eq('user_id', user.id)
            .limit(1);
        
        if (vehicles && vehicles.length > 0) {
            currentVehicle = vehicles[0];
            document.getElementById('vehicleName').innerText = currentVehicle.custom_name;
            active = currentVehicle.is_active;
            updateToggleUI();
        } else {
            // إنشاء مركبة افتراضية إذا لم توجد
            const { data: newVehicle } = await supabase
                .from('vehicles')
                .insert([{ user_id: user.id }])
                .select()
                .single();
            currentVehicle = newVehicle;
            document.getElementById('vehicleName').innerText = 'سيارتي';
        }

        // 4. تحميل الرسائل
        loadMessages();
    }

    function updateToggleUI() {
        const toggle = document.getElementById('statusToggle');
        const label = document.getElementById('statusLabel');
        if (active) {
            toggle.classList.remove('off');
            label.innerHTML = '🟢 حالة الاستقبال';
            document.getElementById('notificationStatus').innerHTML = '📱 سيتم إشعارك فور وصول رسالة';
        } else {
            toggle.classList.add('off');
            label.innerHTML = '🔴 متوقف (خصوصية تامة)';
            document.getElementById('notificationStatus').innerHTML = '🔕 الإشعارات متوقفة';
        }
    }

    async function toggleStatus() {
        if (!currentVehicle) return;
        active = !active;
        updateToggleUI();
        
        const { error } = await supabase
            .from('vehicles')
            .update({ is_active: active })
            .eq('id', currentVehicle.id);
        if (error) alert('فشل تحديث الحالة');
    }

    async function loadMessages() {
        if (!currentVehicle) return;
        
        // جلب الاستيكرات المرتبطة بهذه المركبة
        const { data: stickers } = await supabase
            .from('stickers')
            .select('id')
            .eq('vehicle_id', currentVehicle.id);
        if (!stickers || stickers.length === 0) {
            document.getElementById('inboxContainer').innerHTML = '<div class="loading">لا توجد استيكرات مرتبطة</div>';
            return;
        }
        const stickerIds = stickers.map(s => s.id);
        
        // جلب الرسائل
        const { data: logs, error } = await supabase
            .from('scan_logs')
            .select('*')
            .in('sticker_id', stickerIds)
            .order('created_at', { ascending: false })
            .limit(20);
            
        if (error) {
            document.getElementById('inboxContainer').innerHTML = '<div class="loading">خطأ في التحميل</div>';
            return;
        }
        
        const pendingCount = logs.filter(l => l.status === 'pending').length;
        document.getElementById('newCount').innerText = pendingCount;
        document.getElementById('totalCount').innerText = logs.length;
        
        if (logs.length === 0) {
            document.getElementById('inboxContainer').innerHTML = '<div class="loading">لا توجد رسائل بعد</div>';
            return;
        }
        
        let html = '';
        logs.forEach(log => {
            const date = new Date(log.created_at);
            const timeStr = date.toLocaleString('ar-SA', { hour:'2-digit', minute:'2-digit', day:'numeric', month:'numeric' });
            html += `
                <div class="inbox-item" onclick="selectMessage('${log.id}')">
                    <div class="msg-preview">"${log.visitor_message.substring(0, 50)}${log.visitor_message.length > 50 ? '...' : ''}"</div>
                    <div class="msg-time">${timeStr} · ${log.status === 'pending' ? '🆕 جديدة' : '✅ تم الرد'}</div>
                </div>
            `;
        });
        document.getElementById('inboxContainer').innerHTML = html;
    }

    window.selectMessage = function(messageId) {
        // البحث عن الرسالة المحددة (مخزنة في المتغير المحلي)
        // للتبسيط، سنعيد جلبها
        supabase.from('scan_logs').select('*').eq('id', messageId).single().then(({ data }) => {
            if (data) {
                selectedMessageId = data.id;
                document.getElementById('selectedMessagePreview').innerHTML = `
                    <div class="response-box">
                        <strong>الرسالة:</strong> ${data.visitor_message}<br>
                        ${data.owner_response ? `<strong>ردك السابق:</strong> ${data.owner_response}` : ''}
                    </div>
                `;
                document.getElementById('replyCard').style.display = 'block';
            }
        });
    };

    async function sendReply() {
        const reply = document.getElementById('replyText').value.trim();
        if (!reply || !selectedMessageId) return;
        
        const { error } = await supabase
            .from('scan_logs')
            .update({ owner_response: reply, status: 'closed' })
            .eq('id', selectedMessageId);
            
        if (error) {
            alert('فشل إرسال الرد');
        } else {
            alert('✅ تم إرسال الرد');
            document.getElementById('replyCard').style.display = 'none';
            document.getElementById('replyText').value = '';
            loadMessages();
        }
    }

    window.cancelReply = function() {
        document.getElementById('replyCard').style.display = 'none';
        selectedMessageId = null;
    };

    async function logout() {
        await supabase.auth.signOut();
        window.location.href = '/index.html';
    }

    // بدء التهيئة
    initDashboard();
</script>
