// netlify/functions/redirect.js
const { createClient } = require('@supabase/supabase-js');

const SUPABASE_URL = process.env.SUPABASE_URL;
const SUPABASE_ANON_KEY = process.env.SUPABASE_ANON_KEY;
const supabase = createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

exports.handler = async (event) => {
  // استخراج معرف الاستيكر من نهاية الرابط
  const path = event.path.split('/').pop();
  const stickerToken = path;

  if (!stickerToken) {
    return {
      statusCode: 400,
      body: 'Missing sticker token'
    };
  }

  try {
    // 1. جلب بيانات الاستيكر من قاعدة البيانات
    const { data: sticker, error } = await supabase
      .from('stickers')
      .select('status, vehicle_id')
      .eq('qr_token', stickerToken)
      .single();

    if (error || !sticker) {
      return {
        statusCode: 404,
        body: 'Sticker not found'
      };
    }

    // 2. إذا كان الاستيكر غير مفعّل → صفحة التسجيل
    if (sticker.status === 'factory_new') {
      return {
        statusCode: 302,
        headers: {
          Location: `/register.html?token=${stickerToken}`
        }
      };
    }

    // 3. إذا كان الاستيكر مفعّلاً → التحقق من هوية الزائر
    // نفحص وجود كوكيز الجلسة (سنضيف الكوكيز عند تسجيل الدخول)
    const cookies = event.headers.cookie || '';
    const hasSession = cookies.includes('sb-access-token'); // كوكيز Supabase

    if (hasSession) {
      // إذا كان هناك جلسة نشطة → تحقق مما إذا كان المستخدم هو مالك السيارة
      // (هذا التحقق سيتم داخل dashboard.html نفسها، هنا نوجهه فقط)
      return {
        statusCode: 302,
        headers: {
          Location: `/dashboard.html?token=${stickerToken}`
        }
      };
    }

    // 4. زائر عادي → صفحة التواصل
    return {
      statusCode: 302,
      headers: {
        Location: `/?token=${stickerToken}`
      }
    };
  } catch (err) {
    return {
      statusCode: 500,
      body: 'Internal Server Error'
    };
  }
};
